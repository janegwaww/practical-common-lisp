(defun mystery (x y)
  (if (null y)
      nil
      (if (eql (car y) x)
          0

            (and z (+ z 1))))))

#+begin_src lisp
;; a)
(defun rec-dots (n)
  (if (< n 1)
      nil
      (progn
        (format t ".")
        (rec-dots (- n 1)))))
(defun ite-dots (n)
  (do ((i 0 (+ i 1)))
      ((>= i n) 'done)
    (format t ".")))
#+end_src

#+RESULTS:
: ITE-DOTS

#+begin_src lisp
;; b)
(defun rec-count-a (lst n)
  (if (null lst)
      n
      (if (eql (car lst) 'a)
          (rec-count-a (cdr lst) (+ n 1))
          (rec-count-a (cdr lst) n))))
(defun ite-count-a (lst)
  (let ((n 0))
    (dolist (obj lst)
      (if (eql obj 'a)
          (setf n (+ n 1))))
    n))
#+end_src

#+RESULTS:
: ITE-COUNT-A
